func main() {
        raspiAdaptor := raspi.NewAdaptor()
        oled := spi.NewSSD1306Driver(raspiAdaptor)
        work := func() {
                oled.Clear()
                oled.SetBufferAndDisplay(gobotLogo)
        }
        robot := gobot.NewRobot("ssd1360",
                []gobot.Connection{raspiAdaptor},
                []gobot.Device{oled},
                work,
        )
        robot.Start()
}
